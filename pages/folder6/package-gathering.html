# 二、webpack 配置

配置 webpack 的时候，一般写两份配置文件，一份开发环境，一份生产环境

package.json

{
  "script": {
    "build": "webpack",
  	"dev": "webpack dev --config ./webpack.dev.config.js",
		"prod": "webpack --config ./webpack.prod.config.js"
  }
}

webpack.config.js

const path = require('path')
module.exports = {
	entry: {
  	main: './src/index.js'
  },
  output: {
  	filename: '[name].bundle.js'
  },
  mode: 'production',
}
